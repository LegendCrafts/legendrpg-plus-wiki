攻击力.yml

攻击力.yml

自定义属性配置文件

#属性名称
name: 攻击力
#属性附加值算法,这里可以实现属性默认值的功能,以及属性联动加成功能
additional: '#(力量)# * 2 + 1'
#属性ID,自定义属性后,外部使用该属性的变量,则取这个属性ID值,例如: #(攻击力)#
attributeId: 攻击力
#是否启用
enable: true
#属性图片地址
img: '[local]LegendRpgImage/custom/lr_magicPasv1.png'
#属性描述
info: 
- 属性ID:攻击力
- 属性名:攻击力
#属性LORE增加检测
#特殊变量: #(num)# ,代表装备lore上的数值
lore: '§f攻击力+ #(num)#'
#属性LORE减少检测
remLore: '§f攻击力- #(num)#'
#等级附加值算法,逻辑与属性附加值算法一致
lvAdded: 
#属性等级上限(-1不限)
maxLevel: 1
#属性等级提升一级所需要的属性加点
point: 1
#是否为百分比属性
rate: false
#是否展示到玩家个人属性面板界面中
show: true
#属性排序,展示时,按照该排序进行
sort: 1
#每级属性等级提升的属性值
up: 1.0
#属性升级时执行的指令: (服务器身份执行)
upCmd: 
- 'say #(player)# 属性攻击力升级了'
#属性满级时执行的指令: (服务器身份执行)
upMaxCmd: 
- 'say #(player)# 属性攻击力满级了'
#是否可以加点
upgrade: false
